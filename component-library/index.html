<!-- JavaScript Implementation -->
function initCharacterCounter() {
    const inputCounter = document.getElementById('input-counter');
    const charCounter = inputCounter.parentElement.querySelector('.char-counter');
    
    if (inputCounter && charCounter) {
        const maxLength = inputCounter.getAttribute('maxlength') || 50;
        
        function updateCounter() {
            const currentLength = inputCounter.value.length;
            charCounter.textContent = `${currentLength}/${maxLength}`;
            
            // Visual feedback based on character count
            if (currentLength >= maxLength * 0.9) {
                charCounter.style.color = 'var(--ds-color-error)';
                charCounter.style.fontWeight = 'var(--fw-semibold)';
            } else if (currentLength >= maxLength * 0.7) {
                charCounter.style.color = 'var(--ds-color-primary)';
                charCounter.style.fontWeight = 'var(--fw-medium)';
            } else {
                charCounter.style.color = 'var(--ds-color-gray-60)';
                charCounter.style.fontWeight = 'var(--fw-normal)';
            }
        }
        
        // Event listeners for real-time counting
        inputCounter.addEventListener('input', updateCounter);
        inputCounter.addEventListener('keyup', updateCounter);
        inputCounter.addEventListener('paste', () => setTimeout(updateCounter, 10));
        
        updateCounter(); // Initialize
    }
}

    levelData['Builder-A'], levelData['Enabler-A'],
    levelData['Explorer-A'], levelData['Leader-A']
];
const outerData = [
    levelData['Builder-K'], levelData['Enabler-K'],
    levelData['Explorer-K'], levelData['Leader-K']
];

    chartColors.levels.Builder, chartColors.levels.Enabler,
    chartColors.levels.Explorer, chartColors.levels.Leader
];
const innerColors = outerColors.map(color => color + '80'); // Add transparency

    type: 'doughnut',
    data: {
        labels: innerLabels.concat(outerLabels),
        datasets: [{
            label: 'Level K (Outer)',
            data: outerData,
            backgroundColor: outerColors,
            borderColor: '#ffffff',
            borderWidth: 2,
            weight: 1                    // Controls ring thickness
        }, {
            label: 'Level A (Inner)',
            data: innerData,
            backgroundColor: innerColors,
            borderColor: '#ffffff',
            borderWidth: 2,
            weight: 0.6                  // Thinner inner ring
        }]
    },
    options: {
        responsive: true,
        maintainAspectRatio: false,
        plugins: {
            legend: {
                position: 'right',
                labels: {
                    generateLabels: function(chart) {
                        const labels = [];
                        const datasets = chart.data.datasets;
                        
                        // Generate custom legend showing Level-Type format
                        datasets.forEach((dataset, datasetIndex) => {
                            dataset.data.forEach((value, index) => {
                                if (value > 0) {
                                    const levelType = datasetIndex === 0 ? 'K' : 'A';
                                    const levelName = chart.data.labels[index];
                                    labels.push({
                                        text: `${levelName}-${levelType}: ${value}`,
                                        fillStyle: dataset.backgroundColor[index],
                                        strokeStyle: dataset.borderColor,
                                        lineWidth: dataset.borderWidth,
                                        pointStyle: 'circle'
                                    });
                                }
                            });
                        });
                        return labels;
                    },
                    padding: 10,
                    usePointStyle: true,
                    font: { size: 11 }
                }
            },
            tooltip: {
                callbacks: {
                    title: function(context) {
                        return ''; // Remove title
                    },
                    label: function(context) {
                        const datasetIndex = context.datasetIndex;
                        const levelType = datasetIndex === 0 ? 'K' : 'A';
                        const levelName = context.label;
                        const value = context.parsed;
                        
                        // Calculate percentage within dataset
                        const total = context.dataset.data.reduce((a, b) => a + b, 0);
                        const percentage = ((value / total) * 100).toFixed(1);
                        
                        return `${levelName}-${levelType}: ${value} (${percentage}%)`;
                    }
                },
                backgroundColor: 'rgba(0, 0, 0, 0.8)',
                titleColor: '#ffffff',
                bodyColor: '#ffffff',
                borderColor: 'rgba(255, 255, 255, 0.1)',
                borderWidth: 1
            }
        }
    }
});
